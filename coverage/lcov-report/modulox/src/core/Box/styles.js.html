import { breakpointStyle as generate } from '../../utils'
 
const config = require('../../../config')
 
export const injectStyles = props => {
	const breakpoints = config.breakpoints
	const styles = bp => `
		flex-basis: ${props[bp.size]}%;
		display: ${generate(bp, props.display)};
		position: ${generate(bp, props.position)};
		padding: ${generate(bp, props.padding)};
		width: ${generate(bp, props.width)};
		height: ${generate(bp, props.height)};
		max-width: ${generate(bp, props.maxWidth || `${props[bp.size]}%`)};
		max-height: ${generate(bp, props.maxHeight || `${props[bp.size]}%`)};
		background: ${generate(bp, props.background)};
		margin: ${generate(props.margin)};
		flex-wrap: ${props.wrap ? 'wrap' : 'nowrap'};
		justify-content: ${props.center ? 'center' : generate(bp, props.justify)};
		align-items: ${props.center ? 'center' : generate(bp, props.align)};
		${props.styles}
 
		::after {
			${props.after}
		}
 
		::before {
			${props.before}
		}
 
		:hover {
			${props.hover}
		}
 
		& > * {
			${props.spacing}
		}
	`
 
	return breakpoints.map((bp, index) => {
		if (index === 0) {
			return `${styles(bp)}`;
		}
		return `
			@media (max-width: ${bp.minWidth}px) {
				${styles(bp)}
			}
		`;
	}).join()
}
